<script>
    let schoolClasses = JSON.parse(localStorage.getItem('schoolClasses')) || [];
    let delays = JSON.parse(localStorage.getItem('delays')) || [];
    let editId = null;
    let currentSingleStudent = null;
    let selectedStudentsForReport = [];

    function getSignatureBlock() {
        return `<div class="signature-line"><div class="signature-box"><span>وكيل شؤون الطلاب</span><div class="name">عبدالله المعيبد</div></div><div class="signature-box"><span>مدير المدرسة</span><div class="name">سعد بن عبدالله القرني</div></div></div>`;
    }

    function saveAll() {
        localStorage.setItem('schoolClasses', JSON.stringify(schoolClasses));
        localStorage.setItem('delays', JSON.stringify(delays));
        updateClassSelects();
        renderTable();
        updateStats();
    }

    function saveAllDataManually() {
        saveAll();
        let notification = document.getElementById('saveNotification');
        notification.style.display = 'block';
        setTimeout(() => notification.style.display = 'none', 2000);
        alert("✅ تم حفظ جميع البيانات");
    }

    function addClass() {
        let className = document.getElementById('newClassName').value.trim();
        let studentsText = document.getElementById('studentsListInput').value;
        if(!className) { alert("الرجاء إدخال اسم الفصل"); return; }
        let students = studentsText.split('\n').filter(s => s.trim().length > 0).map(s => s.trim());
        if(students.length === 0) { alert("الرجاء إدخال أسماء الطلاب"); return; }
        let existing = schoolClasses.find(c => c.name === className);
        if(existing) existing.students = [...new Set([...existing.students, ...students])];
        else schoolClasses.push({ name: className, students: students });
        document.getElementById('newClassName').value = '';
        document.getElementById('studentsListInput').value = '';
        saveAll();
        alert(`تم إضافة فصل ${className} بـ ${students.length} طالب`);
    }

    function updateClassSelects() {
        let classSelect = document.getElementById('classSelect');
        classSelect.innerHTML = '<option value="">-- اختر فصل --</option>';
        schoolClasses.forEach(c => classSelect.innerHTML += `<option value="${c.name}">${c.name}</option>`);
    }

    function loadClassStudents() {
        let className = document.getElementById('classSelect').value;
        let studentSelect = document.getElementById('studentSelect');
        studentSelect.innerHTML = '<option value="">-- اختر طالب --</option>';
        if(!className) return;
        let classObj = schoolClasses.find(c => c.name === className);
        if(classObj) classObj.students.forEach(s => studentSelect.innerHTML += `<option value="${s}">${s}</option>`);
    }

    function addDelay() {
        let className = document.getElementById('classSelect').value;
        let studentName = document.getElementById('studentSelect').value;
        let day = document.getElementById('daySelect').value;
        let date = document.getElementById('dateInput').value;
        let minutes = parseInt(document.getElementById('delayMinutes').value);
        if(!className || !studentName || !date || isNaN(minutes)) { alert("يرجى تعبئة جميع الحقول"); return; }
        let exists = delays.some(d => d.name === studentName && d.date === date && d.id !== editId);
        if(exists && editId === null) { alert("⚠️ هذا الطالب مسجل له تأخير في نفس التاريخ!"); return; }
        let hours = minutes / 60;
        if(editId !== null) {
            let index = delays.findIndex(d => d.id === editId);
            if(index !== -1) delays[index] = { id: editId, name: studentName, className, day, date, minutes, hours: hours.toFixed(2) };
            editId = null;
            alert("تم تعديل التأخير");
        } else {
            delays.push({ id: Date.now(), name: studentName, className, day, date, minutes, hours: hours.toFixed(2) });
            alert("تم تسجيل التأخير");
        }
        clearForm();
        saveAll();
    }

    function clearForm() {
        document.getElementById('classSelect').value = '';
        document.getElementById('studentSelect').innerHTML = '<option value="">-- اختر طالب --</option>';
        document.getElementById('delayMinutes').value = '';
        document.getElementById('dateInput').value = '';
        editId = null;
    }

    function editDelay(id) {
        let delay = delays.find(d => d.id === id);
        if(delay) {
            document.getElementById('classSelect').value = delay.className;
            loadClassStudents();
            setTimeout(() => document.getElementById('studentSelect').value = delay.name, 100);
            document.getElementById('daySelect').value = delay.day;
            document.getElementById('dateInput').value = delay.date;
            document.getElementById('delayMinutes').value = delay.minutes;
            editId = id;
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
    }

    function deleteDelay(id) { if(confirm("حذف هذا التأخير؟")){ delays = delays.filter(d => d.id != id); saveAll(); } }
    function deleteAllDelays() { if(confirm("⚠️ حذف جميع التأخيرات بشكل نهائي؟")){ delays = []; saveAll(); } }

    function renderTable() {
        let sortedDelays = [...delays].sort((a,b) => new Date(b.date) - new Date(a.date));
        let tbody = document.getElementById('tableBody');
        if(sortedDelays.length === 0) { tbody.innerHTML = '<tr><td colspan="8">لا توجد تأخيرات مسجلة</td></tr>'; return; }
        tbody.innerHTML = sortedDelays.map((d,idx) => `<tr>
            <td>${idx+1}</td>
            <td>${d.name}</td>
            <td>${d.className}</td>
            <td>${d.day}</td>
            <td>${d.date}</td>
            <td>${d.minutes} دقيقة</td>
            <td>${d.hours} ساعة</td>
            <td><button class="btn btn-warning" style="padding:3px 10px;" onclick="editDelay(${d.id})">✏️ تعديل</button> <button class="btn btn-danger" style="padding:3px 10px;" onclick="deleteDelay(${d.id})">🗑️ حذف</button> <button class="btn btn-info" style="padding:3px 10px;" onclick="showSingleStudentReport('${d.name}')">📄 تقرير</button></td>
        </tr>`).join('');
    }

    function updateStats() {
        let uniqueNames = [...new Set(delays.map(d => d.name))];
        let totalHours = delays.reduce((s,d) => s + parseFloat(d.hours),0);
        let totalRecords = delays.length;
        let studentHours = {};
        delays.forEach(d => { studentHours[d.name] = (studentHours[d.name] || 0) + parseFloat(d.hours); });
        let topStudent = "", topHours = 0;
        for(let name in studentHours) if(studentHours[name] > topHours) { topHours = studentHours[name]; topStudent = name; }
        document.getElementById('totalStudentsStat').innerText = uniqueNames.length;
        document.getElementById('totalHoursStat').innerText = totalHours.toFixed(2);
        document.getElementById('totalRecordsStat').innerText = totalRecords;
        if(topStudent) { document.getElementById('topStudentStat').innerHTML = topStudent; document.getElementById('topStudentHours').innerHTML = topHours.toFixed(2) + " ساعة"; }
        else { document.getElementById('topStudentStat').innerHTML = "-"; document.getElementById('topStudentHours').innerHTML = "0 ساعة"; }
    }

    function showSingleStudentReport(studentName) {
        currentSingleStudent = studentName;
        let studentDelays = delays.filter(d => d.name === studentName);
        let totalHours = studentDelays.reduce((s,d) => s + parseFloat(d.hours),0);
        let className = studentDelays.length > 0 ? studentDelays[0].className : '-';
        let html = `<div style="text-align:center;"><h2>ثانوية النخبة الأهلية</h2><h3>تقرير تأخيرات الطالب</h3></div>
            <div style="background:#f0f7fc; padding:15px; border-radius:20px; margin:20px 0;"><p><strong>اسم الطالب:</strong> ${studentName}</p><p><strong>الفصل:</strong> ${className}</p>
            <p><strong>إجمالي ساعات التأخير:</strong> ${totalHours.toFixed(2)} ساعة</p><p><strong>عدد مرات التأخير:</strong> ${studentDelays.length} مرة</p></div>
            <table style="width:100%; border-collapse:collapse;"><thead><tr style="background:#2c5f8a; color:white;"><th>#</th><th>اليوم</th><th>التاريخ</th><th>مقدار التأخر</th><th>ساعات</th></tr></thead><tbody>`;
        studentDelays.forEach((d,idx) => html += `<tr><td>${idx+1}</td><td>${d.day}</td><td>${d.date}</td><td>${d.minutes} دقيقة</td><td>${d.hours} ساعة</td></tr>`);
        html += `</tbody></table>` + getSignatureBlock();
        document.getElementById('singleStudentReport').innerHTML = html;
        renderTable();
        document.getElementById('singleStudentModal').style.display = 'block';
    }

    function showMultiStudentsPrint() {
        let uniqueStudents = [...new Set(delays.map(d => d.name))];
        if(uniqueStudents.length === 0) { alert("لا يوجد طلاب مسجلين"); return; }
        let html = `<div style="max-height:300px; overflow-y:auto; border:1px solid #ddd; border-radius:20px; padding:15px;">
